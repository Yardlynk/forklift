def exec!(path)
  forklift.logger.log "Running script: #{path}"
  extension = path.split(".").last
  if(extension == "rb" || extension == "ruby")
    exec_ruby(path)
  else
    exec_script(path)
  end
end

def exec_ruby(path)
  klass = forklift.utils.class_name_from_file(path)
  require path
  model = eval("#{klass}.new")
  model.do!(self, forklift)
end
